INTO #emp 
from emppwd t1 WITH(NOLOCK) LEFT JOIN dbo.EmpSysSetting t2 WITH(NOLOCK) ON t1.team=t2.dep AND t2.ntype=2 where t1.dep = '运营部' and team like ('%差旅%') order by t2.depOrder,t1.idnumber
